Creating a Volcano plot

use ggplot and dplyr to filter.

library(circlize)
library(dplyr)
library(tidyverse)
library(ggplot2)
#Creating a volcano plot of logFC vs -log10(P.Value). Adj p values were too high to be used
tmp <- FC.vs.TBS
tmp <- tmp%>%mutate(threshold = ifelse(logFC >= 2, "A", ifelse(logFC <= -2, "B", "C")))
ggplot(tmp, aes(x = logFC, y = -log10(P.Value)))+
         geom_point(aes(colour = threshold), size = 2.5)+
         scale_colour_manual(values = c("A" = "yellow", "B" = "blue", "C" = "black"))+
        geom_hline(aes(yintercept = -log10(0.05)), tmp)+
        geom_vline(xintercept = c(-2,2))
